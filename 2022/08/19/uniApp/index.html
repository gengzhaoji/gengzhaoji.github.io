uniApp vue3
发布日期:   2022-08-19

工程化

目前 uniApp 使用 vue3+vite 的形式进行开发，路由守卫使用 uni-crazy-router，全局共享数据使用 pinia,并封装了 uni.request()请求，以及集成了自动生成 api 的功能；

使用uni-crazy-router

第一步安装：同时安装 regenerator-runtime 是为了解决 uni-crazy-router 运行时的错误；

pnpm i uni-crazy-router regenerator-runtime -S

第二步使用：新建router.js，我放在了utils文件夹下，位置可以自行调整。

import uniCrazyRouter from "uni-crazy-router";

// 接收vue3的实例，并注册uni-crazy-router
export function setupRouter(app) {
  app.use(uniCrazyRouter);
}

/**
 * 白名单
 */
export const whiteList = ["pages/login/login"];

// 路由前置守卫
uniCrazyRouter.beforeEach((to, from, next) => {
  // 判断是否登录， 没有登陆就跳转到登录页， 并且要去除对白名单的拦截， 否则会死循环
  if (!uni.getStorageSync("TOKEN") && !whiteList.includes(to.url)) {
    uniCrazyRouter.afterNotNext(() => {
      // 拦截路由，并且跳转去登录页
      uni.navigateTo({
        url: "/pages/login/login",
      });
    });
    return;
  }
  next();
});

// 路由后置守卫
uniCrazyRouter.afterEach((to, from) => {
  // 逻辑代码
});

// 跳转错误守卫
uniCrazyRouter.onError((to, from) => {
  // 逻辑代码
});

然后再main.js中引入@/utils/router文件，并使用。

import {
	setupRouter
} from '@u/router' // 引入路由

export function createApp() {
	const app = createSSRApp(App);
	// 注册router
	setupRouter(app)
	return {app}
}

使用pinia

第一步安装：pinia 的安装，可以使用HBuilderX插件的安装形式直接安装，或者使用命令安装。

pnpm i pinia -S

第二步使用：在main.js中引入，并使用。

import {
	createSSRApp
} from 'vue'
import * as Pinia from 'pinia'
import App from './App.vue'

export function createApp() {
	const app = createSSRApp(App);
	// 注册Pinia
	app.use(Pinia.createPinia());
	return {
		app,
		Pinia, // 此处必须将 Pinia 返回
	}
}

第三步：创建并使用pinia，store/counter.js，具体使用方法查看pinia

import {
	defineStore
} from 'pinia';

export default defineStore('counter', {
	state: () => ({
		count: 0
	}),
	actions: {
		increment() {
			this.count++;
		},
	},
});

uni.request()请求封装

第一步：需要安装 pnpm i path-to-regexp -S 

/**
 * 通用uni-app网络请求
 * 基于 Promise 对象实现更简单的 request 使用方式，支持请求和响应拦截
 */
import {
	compile
} from 'path-to-regexp';
import {
	getHost
} from '@u/url';

// 请求错误自定义
let errorCode = {
	403: '当前操作没有权限',
	404: '访问资源不存在',
	default: '系统未知错误,请反馈给管理员',
};

// 编译过的url缓存 全局loading
let pathToRegexCaches = {},
	loadingInstance;

// 请求存储
let axiosPromiseArr = [];

// loading白名单
let loadWhite = [];
/**
 * 判断是否为JSON格式
 * @param {*} str
 * @returns
 */
function isJSON(str) {
	if (typeof str == 'string') {
		try {
			JSON.parse(str);
			return true;
		} catch (e) {
			return false;
		}
	}
}

/**
 * 创建缓存key, 由请求url、类型、参数、发送数据构成的标识符
 * @param {string} url 请求url
 * @param {string} type 请求类型
 * @param {object} data 请求数据
 * @return {string}
 */
function createKey({
	url,
	method,
	data = {}
}) {
	return encodeURIComponent([url, method, isJSON(data) ? data : JSON.stringify(data)].join(','));
}

let config = {
		header: {
			'Content-Type': 'application/json;charset=UTF-8'
		},
		data: {},
		method: "GET",
		// 如果设为 json，会尝试对返回的数据做一次 JSON.parse
		dataType: "json",
	},
	// 请求拦截器
	interceptor = {
		/**
		 * 请求前置拦截器（参数处理）
		 */
		request: (config) => {
			// 必须要传入url
			if (!config.url) {
				throw new Error('ajax url is required!');
			}
			// page接口默认添加needCount：1，需要总数
			if (config.url.split('/').pop() === 'page') {
				config.data = Object.assign({
					needCount: 1
				}, config.data || {});
			}
			// export接口默认responseType = 'blob'
			if (config.url.split('/').pop() === 'export') {
				config.responseType = 'blob';
			}

			// 从缓存中提取已经解析过的url
			// url 支持参数信息，如： /api/path/:id
			// 这种情况需要把url解析成一个正则表达式，然后再跟参数匹配组成一个真正要请求的url
			let compileCache = pathToRegexCaches[config.url];

			// 请求的公共url，未设置了完整路径则使用默认路径
			let host = getHost(config.url) || process.env.VUE_APP_BASE_URL;

			if (!compileCache) {
				// 先排除host段，因为host段的端口号与参数写法有冲突
				compileCache = pathToRegexCaches[config.url] = compile(config.url.replace(host, ''), {});
			}
			// 除去传输过来的url参数，并补回host段
			config.url = host + compileCache(config.params);

			// 是否需要设置 token
			const isToken = (config.header || {}).isToken === false,
				token = uni.getStorageSync('TOKEN')
			if (token && !isToken) {
				config.header['Authorization'] = 'Bearer ' + token; // 让每个请求携带自定义token
			}

			// 创建全局loading，排除不需要loading的接口
			if (axiosPromiseArr.length === 0 && !loadWhite.includes(config.url)) uni.showLoading()

			// 判断是否重复
			const key = createKey(config);
			if (axiosPromiseArr.some(item => item === key)) {
				config.repeat = true;
			} else {
				axiosPromiseArr.push(key);
			}
			return config;
		},
		/**
		 * 请求后置拦截器（数据处理）
		 */
		response: (res, resolve, reject) => {
			// 请求状态，未设置则默认为200
			const code = res.data.code || 200;

			// 获取错误信息
			const msg = errorCode[code] || res.data.msg || errorCode['default'];

			if ([401].includes(code)) {
				uni.navigateTo({
					url: '/pages/login/login'
				})
				setTimeout(() => {
					uni.$msg(msg, 'error');
				}, 0)
			} else if (code !== 200) {
				setTimeout(() => {
					uni.$msg(msg, 'error');
				}, 0)
				reject(res.data);
			} else {
				resolve(res.data);
			}
		},
	};

/**
 * 请求方法
 */
export default function(options = {}) {
	// 基于 Promise 的网络请求
	return new Promise((resolve, reject) => {
		/**
		 * 请求前置拦截器
		 */
		let _config = interceptor.request({
			...config,
			...options
		});
		// 请求成功
		_config.success = res => {
			interceptor.response(res, resolve, reject)
		}
		// 请求失败
		_config.fail = error => {
			let {
				message
			} = error;
			if (message) {
				if (message == 'Network Error') {
					message = '后端接口连接异常';
				} else if (message.includes('timeout')) {
					message = '系统接口请求超时';
				} else if (message.includes('Request failed with status code')) {
					message = '系统接口' + message.substr(message.length - 3) + '异常';
				}
				setTimeout(() => {
					uni.$msg(message, 'error', 5 * 1000);
				}, 0)
			}
			reject(error);
		}
		// 接口调用结束的回调函数（调用成功、失败都会执行）
		_config.complete = () => {
			axiosPromiseArr.pop();
			// 全部请求结束关闭loading
			if (axiosPromiseArr.length === 0) uni.hideLoading();
		}
		// 判断是否有正在进行的重复请求
		if (_config.repeat) {
			reject('请求重复！');
		} else {
			uni.request(_config);
		}
	});
}
